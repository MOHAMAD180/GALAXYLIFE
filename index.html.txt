    <script>
      // عملکرد فرم سرمایه‌گذاری
      document.getElementById("investForm").addEventListener("submit", function(e) {
        e.preventDefault();
        var investAmount = parseFloat(document.getElementById("investAmount").value);
        var messageDiv = document.getElementById("investMessage");
        
        // حداقل مبلغ سرمایه‌گذاری 5,000,000 تومان است
        if (investAmount < 5000000) {
          messageDiv.innerHTML = "<div class='alert alert-danger'>حداقل مبلغ سرمایه‌گذاری باید ۵,۰۰۰,۰۰۰ تومان باشد.</div>";
        } else {
          messageDiv.innerHTML = "";
          // مقدار سرمایه‌گذاری وارد شده را در فیلد مبلغ پرداختی قرار می‌دهیم
          document.getElementById("payAmount").value = investAmount;
          // نمایش Modal درگاه پرداخت
          $('#paymentModal').modal('show');
        }
      });

      // عملکرد فرم پرداخت (در Modal)
      document.getElementById("paymentForm").addEventListener("submit", function(e) {
        e.preventDefault();
        // در اینجا می‌توانید منطق پرداخت واقعی را متناسب با API درگاه پرداخت اضافه کنید.
        alert("پرداخت شما با موفقیت انجام شد. متشکریم!");
        // بستن Modal پرداخت
        $('#paymentModal').modal('hide');
        // ریست فرم سرمایه‌گذاری
        document.getElementById("investForm").reset();
      });

      // عملکرد فرم تماس (نمایش پیام ساده)
      document.getElementById("contactForm").addEventListener("submit", function(e) {
        e.preventDefault();
        alert("پیام شما ارسال شد. به زودی به تماس شما پاسخ خواهیم داد.");
        this.reset();
      });
    </script>
